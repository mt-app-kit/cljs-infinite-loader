reload-loader!
(defn reload-loader!
  [loader-id]
  (letfn [(enable-f  [] (enable-loader!  loader-id))
          (disable-f [] (disable-loader! loader-id))]
         (disable-f)
         (time/set-timeout! enable-f 50)))

   [sensor (random/generate-keyword) loader-props])
